    var saved = when.defer(),
        targetDir = this.getTargetDir(config().paths.imagesPath),
        targetFilename;

        targetFilename = filename;
        return nodefn.call(fs.mkdirs, targetDir);

        return nodefn.call(fs.copy, image.path, targetFilename);
